        function updateSessionStats() {
            const days = ['lundi', 'mardi', 'mercredi', 'jeudi', 'vendredi', 'samedi'];

            days.forEach(day => {
                const checkboxes = document.querySelectorAll(`[data-key^="${day}"][data-key$="-done"]`);
                const total = checkboxes.length;
                const done = Array.from(checkboxes).filter(cb => cb.checked).length;

                const exercisesDoneEl = document.getElementById(`${day}-exercises-done`);
                if (exercisesDoneEl) {
                    exercisesDoneEl.textContent = `${done}/${total}`;
                }

                // Calculate volume
                let volume = 0;
                for (let i = 1; i <= total; i++) {
                    const weight = parseFloat(state.data.exercises[`${day}-${i}-weight`]) || 0;
                    const reps = parseFloat(state.data.exercises[`${day}-${i}-reps`]) || 0;
                    volume += weight * reps;
                }

                const volumeEl = document.getElementById(`${day}-volume`);
                if (volumeEl) {
                    volumeEl.textContent = Math.round(volume).toLocaleString();
                }
            });
        }

        function checkForPR(input) {
            const key = input.dataset.key;
            if (!key || !key.includes('-weight')) return;

            const weight = parseFloat(input.value);
            if (isNaN(weight) || weight <= 0) return;

            const exerciseItem = input.closest('.exercise-item');
            if (!exerciseItem) return;

            const exerciseName = exerciseItem.querySelector('.exercise-name')?.textContent;
            if (!exerciseName) return;

            const currentPR = state.data.prs[exerciseName]?.value || 0;

            if (weight > currentPR) {
                state.data.prs[exerciseName] = {
                    value: weight,
                    date: new Date().toISOString().split('T')[0]
                };
                saveData();
                showPRNotification(exerciseName, weight);
                updatePRsDisplay();
            }
        }

        function updateCalendar() {
            const grid = document.getElementById('calendar-grid');
            const monthYearEl = document.getElementById('calendar-month-year');

            if (!grid || !monthYearEl) return;

            const months = ['Janvier', 'Février', 'Mars', 'Avril', 'Mai', 'Juin',
                           'Juillet', 'Août', 'Septembre', 'Octobre', 'Novembre', 'Décembre'];

            monthYearEl.textContent = `${months[state.currentMonth]} ${state.currentYear}`;

            const firstDay = new Date(state.currentYear, state.currentMonth, 1);
            const lastDay = new Date(state.currentYear, state.currentMonth + 1, 0);
            const startDayOfWeek = (firstDay.getDay() + 6) % 7; // Monday = 0

            grid.innerHTML = '';

            // Empty cells before first day
            for (let i = 0; i < startDayOfWeek; i++) {
                const cell = document.createElement('div');
                cell.className = 'calendar-day empty';
                grid.appendChild(cell);
            }

            const today = new Date();

            // Days of month
            for (let day = 1; day <= lastDay.getDate(); day++) {
                const cell = document.createElement('div');
                cell.className = 'calendar-day';

                const dateStr = `${state.currentYear}-${String(state.currentMonth + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
                const dayOfWeek = new Date(state.currentYear, state.currentMonth, day).getDay();

                // Check if today
                if (state.currentYear === today.getFullYear() &&
                    state.currentMonth === today.getMonth() &&
                    day === today.getDate()) {
                    cell.classList.add('today');
                }

                // Check session status
                if (state.data.sessions[dateStr]) {
                    const sessions = Object.values(state.data.sessions[dateStr]);
                    const allCompleted = sessions.every(s => s.completed);
                    if (allCompleted && sessions.length > 0) {
                        cell.classList.add('done');
                    } else if (sessions.length > 0) {
                        cell.classList.add('partial');
                    }
                } else if (dayOfWeek === 0) {
                    // Sunday = rest
                    cell.classList.add('rest');
                }

                cell.innerHTML = `<span class="day-number">${day}</span>`;
                grid.appendChild(cell);
            }
        }

        function setupWaterTracking() {

                        // Fill all up to this

                            bottles[i].classList.add('filled');
                        }
                        state.data.water[today] = bottleNum;
                    }

                    updateWaterDisplay();
                    saveData();
                });
            });

        // ==================== REST TIMER ==================== //
        let timerInterval = null;
        let timerSeconds = 90;
        let timerRunning = false;

        function toggleTimer() {
            const timer = document.getElementById('rest-timer');
            if (timer) {
                timer.style.display = timer.style.display === 'none' ? 'block' : 'none';
            }
        }

        function setTimer(seconds) {
            timerSeconds = seconds;
            timerRunning = false;
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
            updateTimerDisplay();
            const startBtn = document.getElementById('timer-start');
            if (startBtn) startBtn.textContent = '▶️ Start';
        }

        function updateTimerDisplay() {
            const display = document.getElementById('timer-display');
            if (display) {
                const mins = Math.floor(timerSeconds / 60);
                const secs = timerSeconds % 60;
                display.textContent = `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;

                // Animation quand <= 5 secondes
                if (timerSeconds <= 5 && timerRunning) {
                    display.classList.add('timer-warning');
                } else {
                    display.classList.remove('timer-warning');
                }
            }
        }

        function startTimer() {
            const startBtn = document.getElementById('timer-start');

            if (timerRunning) {
                // Pause
                timerRunning = false;
                clearInterval(timerInterval);
                timerInterval = null;
                if (startBtn) startBtn.textContent = '▶️ Start';
            } else {
                // Start
                if (timerSeconds <= 0) {
                    timerSeconds = 90;
                    updateTimerDisplay();
                }
                timerRunning = true;
                if (startBtn) startBtn.textContent = '⏸️ Pause';

                timerInterval = setInterval(() => {
                    timerSeconds--;
                    updateTimerDisplay();

                    if (timerSeconds <= 0) {
                        clearInterval(timerInterval);
                        timerInterval = null;
                        timerRunning = false;
                        if (startBtn) startBtn.textContent = '▶️ Start';
                        playTimerSound();
                        showTimerNotification();
                    }
                }, 1000);
            }
        }

        function resetTimer() {
            timerRunning = false;
            if (timerInterval) {
                clearInterval(timerInterval);
                timerInterval = null;
            }
            timerSeconds = 90;
            updateTimerDisplay();
            const startBtn = document.getElementById('timer-start');
            if (startBtn) startBtn.textContent = '▶️ Start';
        }

        function playTimerSound() {
            try {
                const audioContext = new (window.AudioContext || window.webkitAudioContext)();
                const oscillator = audioContext.createOscillator();
                const gainNode = audioContext.createGain();

                oscillator.connect(gainNode);
                gainNode.connect(audioContext.destination);

                oscillator.frequency.value = 880;
                oscillator.type = 'sine';
                gainNode.gain.setValueAtTime(0.5, audioContext.currentTime);
                gainNode.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.5);

                oscillator.start(audioContext.currentTime);
                oscillator.stop(audioContext.currentTime + 0.5);

                // Double beep
                setTimeout(() => {
                    const osc2 = audioContext.createOscillator();
                    const gain2 = audioContext.createGain();
                    osc2.connect(gain2);
                    gain2.connect(audioContext.destination);
                    osc2.frequency.value = 880;
                    osc2.type = 'sine';
                    gain2.gain.setValueAtTime(0.5, audioContext.currentTime);
                    gain2.gain.exponentialRampToValueAtTime(0.01, audioContext.currentTime + 0.5);
                    osc2.start(audioContext.currentTime);
                    osc2.stop(audioContext.currentTime + 0.5);
                }, 300);
            } catch (e) {
                console.log('Audio non supporté');
            }

            // Vibration sur mobile
            if (navigator.vibrate) {
                navigator.vibrate([200, 100, 200]);
            }
        }

        // ==================== 1RM CALCULATOR ==================== //
        function calculate1RM() {
            const weightInput = document.getElementById('rm-weight');
            const repsInput = document.getElementById('rm-reps');
            const resultEl = document.getElementById('rm-result');

            const weight = parseFloat(weightInput.value);
            const reps = parseInt(repsInput.value);

            if (!weight || !reps || reps < 1) {
                resultEl.textContent = '-- kg';
                document.getElementById('rm-95').textContent = '--';
                document.getElementById('rm-90').textContent = '--';
                document.getElementById('rm-85').textContent = '--';
                document.getElementById('rm-80').textContent = '--';
                document.getElementById('rm-75').textContent = '--';
                document.getElementById('rm-70').textContent = '--';
                return;
            }

            // Formule d'Epley: 1RM = W × (1 + r/30)
            const oneRM = weight * (1 + reps / 30);
            const roundedRM = Math.round(oneRM * 10) / 10;

            resultEl.textContent = roundedRM + ' kg';

            // Calculer les pourcentages
            document.getElementById('rm-95').textContent = Math.round(oneRM * 0.95) + ' kg';
            document.getElementById('rm-90').textContent = Math.round(oneRM * 0.90) + ' kg';
            document.getElementById('rm-85').textContent = Math.round(oneRM * 0.85) + ' kg';
            document.getElementById('rm-80').textContent = Math.round(oneRM * 0.80) + ' kg';
            document.getElementById('rm-75').textContent = Math.round(oneRM * 0.75) + ' kg';
            document.getElementById('rm-70').textContent = Math.round(oneRM * 0.70) + ' kg';
        }
